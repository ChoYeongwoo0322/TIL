```
import sys
sys.stdin = open("의석이.txt")

T = int(input())
for tc in range(1, T+1):
    word_list = []
    for i in range(5):
        words = input()
        word_list.append(words)

    max_len = 0
    for i in word_list:
        if len(i) > max_len:
            max_len = len(i)

    ans = []
    for i in range(max_len):
        for j in word_list:
            if len(j) <= i:
                continue
            ans += j[i]


    print("#{} {}".format(tc, "".join(ans)))
```
