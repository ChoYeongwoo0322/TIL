```
# nPr

def perm(n, k):
    if n == k:
        print(t)
    else:
        for i in range(n):
            if visited[i]: continue  # 방문했으면 건너뛰어라
            t[k] = arr[i]
            visited[i] = 1
            perm(n,k+1) # 하나 높여서 돌리고,
            visited[i] = 0 # 다시 돌아감

arr = [1,2,3]
N = len(arr)
t = [0] * N # 원소의 순서를 저장
visited = [0] * N
perm(N,0)
```
