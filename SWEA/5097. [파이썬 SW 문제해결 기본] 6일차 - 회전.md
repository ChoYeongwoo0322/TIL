```python
import sys

dx=[0,0,-1,1]
dy=[-1,1,0,0]
# 상하좌우, 어떻게 움직일것인지 위치를 설정해준다

def maze(sy,sx):
    global ans # ans를 가져오기
    stack.append((sy,sx)) # 시작값을 stack에 추가
    visited.append((sy,sx)) # 방문했으니까 visited에 추가
# 여기서보면 () 튜플형태로 계속 추가해주나봄
    if arr[sy][sx]==3: # 만약 3이라면? 도착했다는거임
        ans = distance[sy][sx]-1 # ans는 거리에서 -1해주기.  왜해주는지잘모르겠음...
        return # 그리고return

    while len(stack)!=0: # 만약 stack이 비어있지 않다면?
        (sy,sx)=stack.pop(0) # stack의 첫번째값들을 꺼내서 sy,sx로
        for i in range(4): # 방향으로 돌려준다.
            ny=sy+dy[i] # 상 하 좌 우~ 로 돌려~~ 돌려~~
            nx=sx+dx[i]
            if 0<=ny<N and 0<=nx<N and (arr[ny][nx] == 0 or arr[ny][nx] == 3) and (ny,nx) not in visited: # 만약 그 값이 미로내에 포함되어있고, 0이나 3이라면 그리고 아직 방문하지 않았다면?
                stack.append((ny,nx)) # stack에 추가해준다.
                visited.append((ny,nx)) # 방문도 추가해준다.
                distance[ny][nx] = distance[sy][sx]+1 # 그리고 이동한 그 거리좌표에 그 전것보다 +1 해서 한칸 더 움직였다는 것을 보여준다.
                maze(ny,nx) # 그리고 그 움직인 값을 재귀로 돌려준다.
                visited.pop() # 여기까지왔다는것은, 움직일곳이 없다는것 아닐까? 그러면 pop으로 방문한곳 맨마지막 빼주기 pop은 기본값이 -1
                distance[ny][nx] = distance[sy][sx]-1
                # 거리도 -1

    return

for tc in range(1, int(input())+1): # tc
    N = int(input()) # 미로의 크기
    arr=[list(map(int, input())) for _ in range(N)] # 2차행렬
    distance=[[0]*N for _ in range(N)] # 거리를 측정해주기 위해서, 하나하나 체크해줄용도
    visited=[] # 방문했는지검사
    stack = [] # 시작점을 넣어다뺏다하면서 쓸 예정
    ans=0 # 답 = 거리 일것임
    # 출발점 찾기
    for i in range(N):
        for j in range(N):
            if arr[i][j]==2:
                sy,sx=i,j # 시작점찾았다!
                break

    maze(sy,sx) # 이제 함수에 넣어서, 찾으러

    print("#{} {}".format(tc, ans))
```

```python
import sys
sys.stdin = open("subtree_input.txt")

def preorder(node):
    global cnt # 카운트세주기
    if node != 0: # 만약 노드가 0이 아니라면
        cnt += 1                #하고 싶은일, 카운트해주고
        preorder(tree[node][0]) #왼쪽자식 # 왼쪽자식가서 계속 카운트해주기

T = int(input())
for tc in range(1, T+1):
    E, N = map(int, input().split())  # 간선수, 시작노드
    tree = [[0] * 3 for _ in range(E+2)] # 왼쪽, 오른쪽, 부모
    # E+2인 이유는, E가 간선이기때문에. 정점의개수보다 하나 더 작기 때문에, 원래는 정점+1이어야함. 근데 간선이라+1을 더해줌
    temp = list(map(int, input().split()))
    cnt = 0   # 서브트리의 노드의 수
    for i in range(E):
        p = temp[i*2] # 부모
        c = temp[i*2+1] # 자식
        if tree[p][0] == 0: # 만약 왼쪽자식이 비어있다면?
            tree[p][0] = c # 자식을 거기에 넣어주기
        else: # 아니라면
            tree[p][1] = c # 자식을 오른쪽에 넣어주기
        tree[c][2] = p # 그리고 부모는 2번자리에 넣어주고

    preorder(N) # 시작노드넣고 함수로
    print("#{} {}".format(tc, cnt))
```

```python
import sys
sys.stdin=open("16_피자굽기.txt")
from collections import deque

for tc in range(1, int(input())+1):
    n,m = map(int, input().split())
    pizza=list(map(int, input().split()))
    Q = deque()

    for i in range(n):
        Q.append([i,pizza[i]])

    i=0
    while len(Q)!=1:
        Q[0][1] //= 2

        if Q[0][1]==0:
            if n+i<m:
                Q.popleft()
                Q.append([n+i,pizza[n+i]])
                i+=1
            else:
                Q.popleft()
        else:
            Q.append(Q.popleft())

    print("#{} {}".format(tc, Q[0][0]+1))
```

```python
import sys, heapq
sys.stdin=open("17_힙.txt")

for tc in range(1, int(input())+1):
    n = int(input())
    heap=[]
    result = []
    for i in range(n):
        temp = list(map(int, input().split()))

        if temp[0]==1:
            heapq.heappush(heap, -(int(temp[1])))
        elif len(heap):
            result.append(-heapq.heappop(heap))
        else:
            result.append(-1)

    print("#{} {}".format(tc, *result))
```
